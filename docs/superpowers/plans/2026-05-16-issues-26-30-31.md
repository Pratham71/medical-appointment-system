# Issues #26 #30 #31 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. (ask the user)

**Goal:** Three independent features — show all doctors on booking page with availability status (#31), auto-cancel student appointments when a doctor sets an unavailability override (#30), and send an automated emergency alert to infirmary staff from the student dashboard (#26).

**Architecture:** Each feature is a vertical slice: DB query → repository → service → FastAPI route → frontend. #31 adds a read-only endpoint. #30 hooks into the existing override upsert transaction. #26 adds a new alert table and route. All three follow the existing layer pattern: `session.connection_scope()` for reads, `session.transaction_scope()` for writes.

**Tech Stack:** FastAPI, MySQL (mysql-connector-python), Pydantic v2, Next.js 14 App Router, TypeScript, Tailwind CSS, Framer Motion

## Critical patterns — read before touching any backend file

```python
# READ-ONLY repo function pattern (connection_scope):
def my_query(arg: int) -> list[dict[str, Any]]:
    with session.connection_scope() as connection:
        return my_queries.my_query(connection, arg)

# WRITE repo function pattern (transaction_scope):
def my_write(arg: int) -> dict[str, Any] | None:
    with session.transaction_scope() as connection:
        my_queries.my_write(connection, arg)
        return my_queries.get_result(connection, arg)
```

Never use `session.get_connection()` — it does not exist.

def list_doctors_with_availability(
    connection: Any, for_date: date
) -> list[dict[str, Any]]:
    sql = """
        SELECT
            staff.staff_id AS doctor_id,
            users.name AS doctor_name,
            staff.specialization,
            CASE
                WHEN dao.override_id IS NOT NULL
                    THEN IF(dao.is_available, 1, 0)
                WHEN dwa.staff_id IS NOT NULL
                    THEN IF(dwa.is_available, 1, 0)
                ELSE IF(WEEKDAY(%s) < 6, 1, 0)
            END AS is_available,
            CASE
                WHEN dao.override_id IS NOT NULL AND dao.is_available = 0
                    THEN dao.note
                ELSE NULL
            END AS unavailability_note,
            (
                SELECT COUNT(*)
                FROM appointment_slots asl
                INNER JOIN slot_statuses ss
                    ON ss.slot_status_id = asl.slot_status_id
                WHERE asl.staff_id = staff.staff_id
                    AND asl.slot_date = %s
                    AND ss.status_name = 'available'
            ) AS available_slots
        FROM staff
        INNER JOIN users ON users.user_id = staff.user_id
        LEFT JOIN doctor_availability_overrides dao
            ON dao.staff_id = staff.staff_id
            AND dao.override_date = %s
        LEFT JOIN doctor_weekly_availability dwa
            ON dwa.staff_id = staff.staff_id
            AND dwa.weekday = WEEKDAY(%s)
        WHERE staff.is_doctor = TRUE
        ORDER BY users.name
    """
    return fetch_all(connection, sql, (for_date, for_date, for_date, for_date))

def list_appointments_to_cancel(
    connection: Any,
    staff_id: int,
    override_date: date,
    start_time: time | None,
    end_time: time | None,
) -> list[dict[str, Any]]:
    """Return booked appointment_ids for a doctor on a date, optionally within a time window."""
    if start_time and end_time:
        sql = """
            SELECT appointments.appointment_id, appointments.student_id,
                   appointment_slots.slot_id
            FROM appointments
            INNER JOIN appointment_slots
                ON appointment_slots.slot_id = appointments.slot_id
            INNER JOIN appointment_statuses
                ON appointment_statuses.status_id = appointments.status_id
            WHERE appointment_slots.staff_id = %s
                AND appointment_slots.slot_date = %s
                AND appointment_slots.start_time >= %s
                AND appointment_slots.end_time <= %s
                AND appointment_statuses.status_name = 'booked'
        """
        return fetch_all(connection, sql, (staff_id, override_date, start_time, end_time))
    else:
        sql = """
            SELECT appointments.appointment_id, appointments.student_id,
                   appointment_slots.slot_id
            FROM appointments
            INNER JOIN appointment_slots
                ON appointment_slots.slot_id = appointments.slot_id
            INNER JOIN appointment_statuses
                ON appointment_statuses.status_id = appointments.status_id
            WHERE appointment_slots.staff_id = %s
                AND appointment_slots.slot_date = %s
                AND appointment_statuses.status_name = 'booked'
        """
        return fetch_all(connection, sql, (staff_id, override_date))


def cancel_appointment_with_reason(
    connection: Any,
    appointment_id: int,
    cancelled_status_id: int,
    available_slot_status_id: int,
    slot_id: int,
    cancellation_reason: str,
) -> None:
    execute(
        connection,
        "UPDATE appointments SET status_id = %s, cancellation_reason = %s WHERE appointment_id = %s",
        (cancelled_status_id, cancellation_reason, appointment_id),
    )
    execute(
        connection,
        "UPDATE appointment_slots SET slot_status_id = %s WHERE slot_id = %s",
        (available_slot_status_id, slot_id),
    )

def upsert_availability_override(
    staff_id: int,
    override_date: date,
    is_available: bool,
    start_time: time | None,
    end_time: time | None,
    note: str | None,
) -> dict[str, Any] | None:
    with session.transaction_scope() as connection:
        doctor_queries.upsert_availability_override(
            connection,
            staff_id=staff_id,
            override_date=override_date,
            is_available=is_available,
            start_time=start_time,
            end_time=end_time,
            note=note,
        )

        if not is_available:
            cancelled_status_id = appointment_queries.get_appointment_status_id(
                connection, "cancelled"
            )
            available_slot_status_id = appointment_queries.get_slot_status_id(
                connection, "available"
            )
            if cancelled_status_id and available_slot_status_id:
                reason = f"Doctor unavailable{': ' + note if note else ''}"
                affected = appointment_queries.list_appointments_to_cancel(
                    connection,
                    staff_id=staff_id,
                    override_date=override_date,
                    start_time=start_time,
                    end_time=end_time,
                )
                for appt in affected:
                    appointment_queries.cancel_appointment_with_reason(
                        connection,
                        appointment_id=appt["appointment_id"],
                        cancelled_status_id=cancelled_status_id,
                        available_slot_status_id=available_slot_status_id,
                        slot_id=appt["slot_id"],
                        cancellation_reason=reason,
                    )

        return doctor_queries.get_availability_override(connection, staff_id, override_date)
